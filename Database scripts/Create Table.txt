CREATE TABLE PRODUCT
(
  IDPRODUCT INTEGER NOT NULL,
  IDMANUFACTURER INTEGER NOT NULL,
  IDPRODUCTFORM INTEGER DEFAULT 0,
  PRODUCTNAME VARCHAR(50),
  PRODUCTPACK VARCHAR(9),
  PRODUCTSIZE CHAR(10) DEFAULT '',
  PRODUCTDESCRIPTION VARCHAR(100),
  PRODUCTBARCODE VARCHAR(20),
  PRODUCTNAPPICODE VARCHAR(12),
  PRODUCTSCHEDULE VARCHAR(1),
  DISCONTINUED VARCHAR(1) DEFAULT 'N',
  LASTCHANGED TIMESTAMP,
  PRODUCTSTRENGTH VARCHAR(15),
  PRODUCTTYPE VARCHAR(1),
  CATALOGUENUMBER VARCHAR(30),
  DOSAGEFORM VARCHAR(5),
  NAMEUSED VARCHAR(1),
  CONSTRAINT PK_PRODUCT_0 PRIMARY KEY (IDPRODUCT)
);



CREATE TABLE STOCKITEM
(
  IDSTOCKITEM INTEGER NOT NULL,
  IDPRODUCT INTEGER NOT NULL,
  IDSUPPLIER INTEGER NOT NULL,
  STOCKITEMPRODNO VARCHAR(16),
  STOCKITEMNAME VARCHAR(50),
  STOCKITEMPACK VARCHAR(5),
  STOCKITEMSELLINGPRICE DECIMAL(12,2),
  STOCKITEMSOH INTEGER,
  STOCKITEMAGENTOUT VARCHAR(1),
  EXPDATE1 DATE,
  EXPDATE2 DATE,
  EXPDATE3 DATE,
  SELLINGMODULE VARCHAR(10),
  DISCONTINUED VARCHAR(1) DEFAULT 'N',
  LASTCHANGED TIMESTAMP,
  CREATEDATE DATE,
  CONSTRAINT INTEG_132 PRIMARY KEY (IDSTOCKITEM)
);
